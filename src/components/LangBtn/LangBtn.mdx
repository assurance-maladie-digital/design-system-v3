# Exemple d'utilisation

<Source dark code={`
<script setup lang="ts">
 import LangBtn from '@/components/LangBtn/LangBtn.vue'
 import { ref } from 'vue'

 const value = ref('fr')

 const ariaLabel = 'Choix de la langue'
</script>

<template>
 <div class="mt-12 ml-12">
  <LangBtn
   v-model="value"
   :available-languages="items"
   :aria-label="ariaLabel"
   hide-down-arrow
  />
 </div>
</template>
`} />
